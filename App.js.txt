import { useState } from "react";
import { ethers } from "ethers";
import { CONTRACT_ADDRESS, CONTRACT_ABI } from "./config";

function App() {
  const [opcion, setOpcion] = useState(0);
  const [monto, setMonto] = useState("");
  const [reporte, setReporte] = useState(null);

  async function getContract() {
    const provider = new ethers.BrowserProvider(window.ethereum);
    const signer = await provider.getSigner();
    return new ethers.Contract(CONTRACT_ADDRESS, CONTRACT_ABI, signer);
  }

  async function apostar() {
    const contract = await getContract();
    const tx = await contract.apostar(opcion, { value: ethers.parseEther(monto) });
    await tx.wait();
    alert("Apuesta realizada!");
  }

  async function cerrarApuestas() {
    const contract = await getContract();
    const tx = await contract.cerrarApuestas();
    await tx.wait();
    alert("Apuestas cerradas!");
  }

  async function obtenerReporte() {
    const contract = await getContract();
    const resultado = await contract.reporteFinal();

    setReporte({
      resultado: resultado[0],
      todosApostadores: resultado[1],
      ganadores: resultado[2],
      pagos: resultado[3],
      fondo: resultado[4]
    });
  }

  return (
    <div>
      <h1>DApp de Apuestas</h1>

      <h2>Registrar apuesta</h2>
      <input type="number" placeholder="Opción" onChange={(e) => setOpcion(e.target.value)} />
      <input type="text" placeholder="Monto en ETH" onChange={(e) => setMonto(e.target.value)} />
      <button onClick={apostar}>Apostar</button>

      <h2>Admin</h2>
      <button onClick={cerrarApuestas}>Cerrar Apuestas</button>

      <h2>Reporte</h2>
      <button onClick={obtenerReporte}>Ver Reporte Final</button>

      {reporte && (
        <div>
          <h3>Resultado: {reporte.resultado}</h3>
          <h3>Fondo de continuidad: {ethers.formatEther(reporte.fondo)} ETH</h3>
          <h3>Ganadores:</h3>
          <ul>
            {reporte.ganadores.map((g, idx) => (
              <li key={g}>
                {g} → {ethers.formatEther(reporte.pagos[idx])} ETH
              </li>
            ))}
          </ul>
        </div>
      )}
    </div>
  );
}

export default App;
